84// returns number of bytes received (in the UART rx fifo buffer)
85int SerialPort::uartGetRXFIFOCount()
86{
87 return m_dev->status.rxfifo_cnt | ((int)(m_dev->mem_cnt_status.rx_cnt) << 8);
88}

92void SerialPort::uartFlushTXFIFO()

94 while (m_dev->status.txfifo_cnt || m_dev->status.st_utx_out)

99// flushes RX buffer of UART
100void SerialPort::uartFlushRXFIFO()
101{
102 while (uartGetRXFIFOCount() != 0 || m_dev->mem_rx_status.wr_addr != m_dev->mem_rx_status.rd_addr)
103 m_dev->fifo.rw_byte;
104}

126{

130
131 // TX (out)
132 m_txPin = int2gpio(txPin);
133 configureGPIO(m_txPin, GPIO_MODE_OUTPUT);
134
135 // RTS (out)

140 }
141
142 // DTR (out)

144 if (m_dtrPin != GPIO_UNUSED) {
145 configureGPIO(m_dtrPin, GPIO_MODE_OUTPUT);
146 setDTRStatus(true); // assert DTR
147 }
148
149 // CTS (in)
150 m_ctsPin = int2gpio(ctsPin);
151 if (m_ctsPin != GPIO_UNUSED) {
152 configureGPIO(m_ctsPin, GPIO_MODE_INPUT);
153 }
154
155 // DSR (in)
156 m_dsrPin = int2gpio(dsrPin);
157 if (m_dsrPin != GPIO_UNUSED) {
158 configureGPIO(m_dsrPin, GPIO_MODE_INPUT);
159 }
160
161 // DCD (in)

175// uart = 0, 1, 2
176void SerialPort::setup(int uartIndex, uint32_t baud, int dataLength, char parity, float stopBits, FlowControl flowControl, bool inverted)
177{
178 if (!m_initialized) {
179 // uart not configured, configure now
180
181 m_idx = uartIndex;
182 m_dev = (volatile uart_dev_t *) UART_BASE[m_idx];
183
184 switch (m_idx) {
185 case 0:
186 #ifdef ARDUINO
187 Serial.end();
188 #endif
189 DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_UART_CLK_EN);
190 DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_UART_RST);
191 break;
192 case 1:
193 #ifdef ARDUINO
194 Serial1.end();
195 #endif
196 DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_UART1_CLK_EN);
197 DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_UART1_RST);
198 break;
199 case 2:
200 #ifdef ARDUINO
201 Serial2.end();
202 #endif
203 DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_UART2_CLK_EN);
204 DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_UART2_RST);
205 break;
206 }
207
208 // flush
209 uartFlushTXFIFO();
210 uartFlushRXFIFO();
211
212 // setup RX interrupt
213 WRITE_PERI_REG(UART_CONF1_REG(m_idx), (1 << UART_RXFIFO_FULL_THRHD_S) | // an interrupt for each character received
214 (2 << UART_RX_TOUT_THRHD_S) | // actually not used
215 (0 << UART_RX_TOUT_EN_S)); // timeout not enabled
216 WRITE_PERI_REG(UART_INT_ENA_REG(m_idx), (1 << UART_RXFIFO_FULL_INT_ENA_S) | // interrupt on FIFO full (1 character - see rxfifo_full_thrhd)
217 (1 << UART_FRM_ERR_INT_ENA_S) | // interrupt on frame error
218 (0 << UART_RXFIFO_TOUT_INT_ENA_S) | // no interrupt on rx timeout (see rx_tout_en and rx_tout_thrhd)
219 (1 << UART_PARITY_ERR_INT_ENA_S) | // interrupt on rx parity error
220 (1 << UART_RXFIFO_OVF_INT_ENA_S) | // interrupt on rx overflow
221 (1 << UART_BRK_DET_INT_ENA_S)); // interrupt on Break
222 WRITE_PERI_REG(UART_INT_CLR_REG(m_idx), 0xffffffff);
223 esp_intr_alloc_pinnedToCore(INTR_SRC[m_idx], 0, uart_isr, this, nullptr, CoreUsage::quietCore());
224
225 // setup FIFOs size
226 WRITE_PERI_REG(UART_MEM_CONF_REG(m_idx), (3 << UART_RX_SIZE_S) | // RX: 3 * 128 = 384 bytes (this is the max for UART2)
227 (1 << UART_TX_SIZE_S)); // TX: 1 * 128 = 128 bytes
228 }
229
230 m_flowControl = flowControl;
231
232 m_inverted = inverted;
233
234 // baud rate
235 setBaud(baud);
236
237 // frame
238 setFrame(dataLength, parity, stopBits);
239
240 // TX/RX Pin logic
241 gpio_matrix_in(m_rxPin, URXD_IN_IDX[m_idx], m_inverted);
242 gpio_matrix_out(m_txPin, UTXD_OUT_IDX[m_idx], m_inverted, false);
243
244 // Flow Control
245 WRITE_PERI_REG(UART_FLOW_CONF_REG(m_idx), 0);
246
247 m_initialized = true;
248
249 // APB Change callback (TODO?)
250 //addApbChangeCallback(this, uart_on_apb_change);
251}

254void SerialPort::setBaud(int value)
255{
256 uint32_t clk_div = (getApbFrequency() << 4) / value;
257 WRITE_PERI_REG(UART_CLKDIV_REG(m_idx), ((clk_div >> 4) << UART_CLKDIV_S) |
258 ((clk_div & 0xf) << UART_CLKDIV_FRAG_S));
259}

262void SerialPort::setFrame(int dataLength, char parity, float stopBits)
263{
264 uint32_t config0 = (1 << UART_TICK_REF_ALWAYS_ON_S) | ((dataLength - 5) << UART_BIT_NUM_S);
265 if (parity == 'E')
266 config0 |= (1 << UART_PARITY_EN_S);
267 else if (parity == 'O')
268 config0 |= (1 << UART_PARITY_EN_S) | (1 << UART_PARITY_S);
269 if (stopBits == 1.0)
270 config0 |= 1 << UART_STOP_BIT_NUM_S;
271 else if (stopBits == 1.5)
272 config0 |= 2 << UART_STOP_BIT_NUM_S;
273 else if (stopBits >= 2.0) {
274 config0 |= 1 << UART_STOP_BIT_NUM_S;
275 SET_PERI_REG_BITS(UART_RS485_CONF_REG(m_idx), UART_DL1_EN_V, 1, UART_DL1_EN_S); // additional 1 stop bit
276 if (stopBits >= 3.0)
277 SET_PERI_REG_BITS(UART_RS485_CONF_REG(m_idx), UART_DL0_EN_V, 1, UART_DL1_EN_S); // additional 1 stop bit
278 }
279 WRITE_PERI_REG(UART_CONF0_REG(m_idx), config0);
280}

283// enable/disable RX sending XON/XOFF and/or setting RTS
284void SerialPort::flowControl(bool enableRX)
285{
286 bool isrEnabled = m_dev->int_ena.rxfifo_full;
287 CLEAR_PERI_REG_MASK(UART_INT_ENA_REG(m_idx), UART_RXFIFO_FULL_INT_ENA_M);
288
289 if (enableRX) {
290 // resume RX
291 if (m_sentXOFF) {
292 if (m_flowControl == FlowControl::Software || m_flowControl == FlowControl::Hardsoft)
293 send(ASCII_XON);
294 if (m_flowControl == FlowControl::Hardware || m_flowControl == FlowControl::Hardsoft)
295 setRTSStatus(true); // assert RTS
296 isrEnabled = true;
297 m_sentXOFF = false;
298 }
299 } else {
300 // suspend RX
301 if (!m_sentXOFF) {
302 if (m_flowControl == FlowControl::Software || m_flowControl == FlowControl::Hardsoft)
303 send(ASCII_XOFF);
304 if (m_flowControl == FlowControl::Hardware || m_flowControl == FlowControl::Hardsoft)
305 setRTSStatus(false); // disable RTS
306 // note: here we don't disable FIFO interrupts. These are disabled in ISR when input queue is full
307 m_sentXOFF = true;
308 }
309 }
310
311 if (isrEnabled)
312 SET_PERI_REG_MASK(UART_INT_ENA_REG(m_idx), UART_RXFIFO_FULL_INT_ENA_M); // enable RX FIFO full interrupt
313}

316// check if TX is enabled looking for XOFF received or reading CTS
317bool SerialPort::readyToSend()
318{
319 //Serial.printf("readyToSend m_recvXOFF=%d\n", m_recvXOFF);
320 if ((m_flowControl == FlowControl::Software || m_flowControl == FlowControl::Hardsoft) && m_recvXOFF)
321 return false; // TX disabled (received XOFF)
322 if ((m_flowControl == FlowControl::Hardware || m_flowControl == FlowControl::Hardsoft) && CTSStatus() == false)
323 return false; // TX disabled (CTS=high, not active)
324 return true; // TX enabled
325}

336void SerialPort::sendBreak(bool value)
337{
338 constexpr int MATRIX_DETACH_OUT_SIG = 0x100;
339
340 while (m_dev->status.txfifo_cnt == 0x7F)
341 ;
342 if (value) {
343 gpio_matrix_out(m_txPin, MATRIX_DETACH_OUT_SIG, m_inverted, false);
344 configureGPIO(m_txPin, GPIO_MODE_OUTPUT);
345 gpio_set_level(m_txPin, 0);
346 } else {
347 gpio_matrix_out(m_txPin, UTXD_OUT_IDX[m_idx], m_inverted, false);
348 }
349}

352void IRAM_ATTR SerialPort::uart_isr(void * arg)
353{
354 auto ser = (SerialPort*) arg;
355 auto dev = ser->m_dev;
356 auto idx = ser->m_idx;
357
358 // look for overflow or RX errors
359 if (dev->int_st.rxfifo_ovf || dev->int_st.frm_err || dev->int_st.parity_err) {
360 // save (or) error states
361 ser->m_parityError |= dev->int_st.parity_err;
362 ser->m_framingError |= dev->int_st.frm_err;
363 ser->m_overflowError |= dev->int_st.rxfifo_ovf;
364 if (ser->m_lineStatusCallback)
365 ser->m_lineStatusCallback(ser->m_callbackArgs, ser->m_parityError, ser->m_framingError, ser->m_overflowError, ser->m_breakDetected, true);
366 // reset RX-FIFO, because hardware bug rxfifo_rst cannot be used, so just flush
367 ser->uartFlushRXFIFO();
368 // clear interrupt flags
369 SET_PERI_REG_MASK(UART_INT_CLR_REG(idx), UART_RXFIFO_OVF_INT_CLR_M | UART_FRM_ERR_INT_CLR_M | UART_PARITY_ERR_INT_CLR_M);
370 return;
371 }
372
373 // Break detection
374 if (dev->int_st.brk_det) {
375 // save (or) break state
376 ser->m_breakDetected |= dev->int_st.brk_det;
377 if (ser->m_lineStatusCallback)
378 ser->m_lineStatusCallback(ser->m_callbackArgs, ser->m_parityError, ser->m_framingError, ser->m_overflowError, ser->m_breakDetected, true);
379 // clear interrupt flags
380 SET_PERI_REG_MASK(UART_INT_CLR_REG(idx), UART_BRK_DET_INT_CLR_M);
381 return;
382 }
383
384 // flow control?
385 if (ser->m_flowControl != FlowControl::None) {
386 // send XOFF/XON or set RTS looking at RX FIFO occupation
387 int count = ser->uartGetRXFIFOCount();
388 if (count > FABGLIB_TERMINAL_FLOWCONTROL_RXFIFO_MAX_THRESHOLD)
389 ser->flowControl(false); // suspend RX
390 else if (count < FABGLIB_TERMINAL_FLOWCONTROL_RXFIFO_MIN_THRESHOLD)
391 ser->flowControl(true); // resume RX
392 }
393
394 // main receive loop
395 while (ser->uartGetRXFIFOCount() != 0 || dev->mem_rx_status.wr_addr != dev->mem_rx_status.rd_addr) {
396 // look for enough room in input queue
397 if (ser->m_flowControl != FlowControl::None && !ser->m_rxReadyCallback(ser->m_callbackArgs, true)) {
398 ser->flowControl(false); // suspend RX
399 // block further interrupts
400 CLEAR_PERI_REG_MASK(UART_INT_ENA_REG(idx), UART_RXFIFO_FULL_INT_ENA_M);
401 break;
402 }
403 // get char
404 uint8_t r = READ_PERI_REG(UART_FIFO_REG(idx));
405 // XON/XOFF?
406 if ((r == ASCII_XOFF || r == ASCII_XON) && (ser->m_flowControl == FlowControl::Software || ser->m_flowControl == FlowControl::Hardsoft)) {
407 ser->m_recvXOFF = (r == ASCII_XOFF);
408 } else {
409 // add to input queue
410 ser->m_rxCallback(ser->m_callbackArgs, r, true);
411 }
412 }
413
414 // clear interrupt flag
415 SET_PERI_REG_MASK(UART_INT_CLR_REG(idx), UART_RXFIFO_FULL_INT_CLR_M);
416}

439void SerialPortTerminalConnector::connect(SerialPort * serialPort, Terminal * terminal)
440{
441 m_serialPort = serialPort;
442 m_serialPort->setCallbacks(this, rxReadyCallback, rxCallback);
443
444 m_terminal = terminal;
445 m_terminal->connectKeyboard();
446
447 m_terminal->onReceive = [&](uint8_t c) {
448 // resume RX?
449 if (!m_serialPort->readyToReceive() && m_terminal->availableForWrite())
450 m_serialPort->flowControl(true);
451 };
452
453 m_terminal->onReadyToSend = [&](bool * readyToSend) {
454 // suspend/resume TX?
455 if (*readyToSend)
456 *readyToSend = m_serialPort->readyToSend();
457 };
458
459 m_terminal->onSend = [&](uint8_t c) {
460 m_serialPort->send(c);
461 };
462}
