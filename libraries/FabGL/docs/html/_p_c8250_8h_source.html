117 // Transmitter Holding Register (THR)
118 uint8_t m_THR;
119
120 // Interrupt Identification Register (IIR)
121 uint8_t m_IIR;
122
123 // Interrupt Enable Register (IER)
124 // 0 = Receiver Data Available interrupt enabled
125 // 1 = Transmitter Holding Register Empty interrupt enabled
126 // 2 = Receiver Line Status interrupt enabled
127 // 3 = MODEM Status interrupt enabled
128 uint8_t m_IER;
129
130 // interrupt triggered flags (not an actual 8250 register)
131 // 0 = Receiver Data Available interrupt triggered
132 // 1 = Transmitter Holding Register Empty interrupt triggered
133 // 2 = Receiver Line Status interrupt triggered
134 // 3 = MODEM Status interrupt triggered
135 uint8_t m_trigs;
136
137 // Line Control Register (LCR):
138 // 0, 1 = number of bits: 00 = 5 bits, 01 = 6 bits, 10 = 7 bits, 11 = 8 bits
139 // 2 = number of stop bits: 0 = 1 stop bit, 1 = if 5 bits then 1.5 stop bits, 2 stop bits otherwise
140 // 3 = parity enable: 1 = parity enabled
141 // 4 = parity type: 0 = odd, 1 = even
142 // 5 = stick parity (UNSUPPORTED)
143 // 6 = send break
144 // 7 = Divisor Latch Access Bit (DLAB)
145 uint8_t m_LCR;
146
147 // MODEM Control Register (MCR)
148 // 0 = DTR out (connected to DSR in loopback mode)
149 // 1 = RTS out (connected to CTS in loopback mode)
150 // 2 = OUT1 out (connected to RI in loopback mode)
151 // 3 = OUT2 out (connected to DCD in loopback mode)
152 // 4 = loopback mode (1 = enabled)
153 uint8_t m_MCR;
154
155 // Line Status Register (LSR)
156 // 0 = Data Ready (DR), 1 = data available
157 // 1 = Overrun Error (OE), 1 = error
158 // 2 = Parity Error (PE), 1 = error
159 // 3 = Framing Error (FE), 1 = error
160 // 4 = Break Interrupt (BI), 1 = break received
161 // 5 = Transmitter Holding Register Empty (THRE), 1 = transmitter holding register empty
162 // 6 = Transmitter Empty indicator (TEMT), 1 = transmitter shift register empty
163 uint8_t m_LSR;
164
165 // MODEM Status Register (MSR)
166 // 0 = Delta CTS (DCTS), 1 = CTS changed since last read
167 // 1 = Delta DSR (DDSR), 1 = DSR changed since last read
168 // 2 = Trailing Edge of RI (TERI), 1 = RI has changed from hi to low
169 // 3 = Delta DCD (DDCD), 1 = DCD changed since last read
170 // 4 = CTS in status (connected to RTS in loopback mode)
171 // 5 = DSR in status (connected to DTR in loopback mode)
172 // 6 = RI in status (connected to OUT1 in loopback mode)
173 // 7 = DCD in status (connected to OUT2 in loopback mode)
174 uint8_t m_MSR;
175
176 // Scratch Register (SCR)
177 uint8_t m_SCR;
178
179 // Divisor Latch low byte (DLL)
180 uint8_t m_DLL;
181
182 // Divisor Latch high byte (DLM)
183 uint8_t m_DLM;
184
185 // bytes lost flag (not an actual 8250 register)
186 volatile bool m_overrun;
187
188 // a byte has been received
189 volatile bool m_dataReady;
190 volatile uint8_t m_dataRecv;
191
192};
